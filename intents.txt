Начать игру
start_game

slots:
    game:
        source: $Game
        type: Game
    word:
        source: $Word
        type: Word
$Start:
    %lemma
    давай
    начать
    начинать
    начинаться
    угадать
    спрашивать
    сыграем
    поиграем
    жги
    старт
    стартовать
    игра
    играть
    да
    продолжить
root:
    [($Start)+ ($Game)* ($Word)*] | [($Start)* ($Game)+ | ($Word)+]
filler:
    алиса
    еще
